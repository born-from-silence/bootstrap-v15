import logging
from jira import JIRA, JIRAError
from requests.exceptions import RequestException, Timeout, ConnectionError
from functools import wraps

logger = logging.getLogger(__name__)

class JiraAPIError(Exception):
    """Custom exception for Jira API errors"""
    pass

class JiraNotFoundError(JiraAPIError):
    """When a resource doesn't exist"""
    pass

class JiraPermissionError(JiraAPIError):
    """When access is denied"""
    pass

class JiraValidationError(JiraAPIError):
    """When data validation fails"""
    pass

def handle_jira_errors(func):
    """Decorator for centralized error handling"""
    @wraps(func)
    def wrapper(*args, **kwargs):
        try:
            return func(*args, **kwargs)
        except JIRAError as e:
            # Map JIRA error codes to custom exceptions
            if e.status_code == 404:
                logger.error(f"Resource not found: {e.text}")
                raise JiraNotFoundError(f"Jira resource not found: {e.text}")
            elif e.status_code == 403:
                logger.error(f"Permission denied: {e.text}")
                raise JiraPermissionError(f"Access denied: {e.text}")
            elif e.status_code == 400:
                logger.error(f"Validation error: {e.text}")
                raise JiraValidationError(f"Invalid data: {e.text}")
            else:
                logger.error(f"Jira API error ({e.status_code}): {e.text}")
                raise JiraAPIError(f"Jira API error: {e.text}")
        except Timeout:
            logger.error("Request timed out")
            raise JiraAPIError("Request to Jira API timed out")
        except ConnectionError:
            logger.error("Connection failed")
            raise JiraAPIError("Failed to connect to Jira server")
        except RequestException as e:
            logger.error(f"Request error: {e}")
            raise JiraAPIError(f"Request failed: {e}")
    return wrapper

import logging
import sys
from pathlib import Path

def setup_logging(
    log_level: int = logging.INFO,
    log_file: str = "jira_operations.log",
    console_output: bool = True
) -> logging.Logger:
    """
    Setup comprehensive logging for Jira operations
    
    Args:
        log_level: Minimum logging level
        log_file: Path to log file
        console_output: Whether to output to console
    
    Returns:
        Configured logger instance
    """
    logger = logging.getLogger("jira_client")
    logger.setLevel(log_level)
    
    # Clear existing handlers
    logger.handlers = []
    
    # Create formatter
    formatter = logging.Formatter(
        '%(asctime)s | %(levelname)-8s | %(name)s | %(funcName)s:%(lineno)d | %(message)s',
        datefmt='%Y-%m-%d %H:%M:%S'
    )
    
    # File handler with rotation
    from logging.handlers import RotatingFileHandler
    file_handler = RotatingFileHandler(
        log_file,
        maxBytes=10*1024*1024,  # 10MB
        backupCount=5,
        encoding='utf-8'
    )
    file_handler.setLevel(logging.DEBUG)
    file_handler.setFormatter(formatter)
    logger.addHandler(file_handler)
    
    # Console handler
    if console_output:
        console_handler = logging.StreamHandler(sys.stdout)
        console_handler.setLevel(log_level)
        console_handler.setFormatter(formatter)
        logger.addHandler(console_handler)
    
    return logger

# Usage

from enum import Enum, auto
from dataclasses import dataclass
from typing import Optional, List, Dict

class IssueType(Enum):
    """Standard Jira issue types"""
    BUG = "Bug"
    TASK = "Task"
    STORY = "Story"
    EPIC = "Epic"
    SUBTASK = "Sub-task"
    IMPROVEMENT = "Improvement"
    NEW_FEATURE = "New Feature"
    
    @classmethod
    def from_string(cls, value: str) -> "IssueType":
        """Create enum from string, case-insensitive"""
        try:
            return cls(value.upper().replace("-", "_"))
        except ValueError:
            # Fallback for custom types
            for member in cls:
                if member.value.lower() == value.lower():
                    return member
            raise ValueError(f"Unknown issue type: {value}")

class Priority(Enum):
    """Jira priorities"""
    HIGHEST = 1
    HIGH = 2
    MEDIUM = 3
    LOW = 4
    LOWEST = 5
    
    def __str__(self) -> str:
        return self.name

class StatusCategory(Enum):
    """High-level status categories"""
    TODO = "To Do"
    IN_PROGRESS = "In Progress" 
    DONE = "Done"

@dataclass
class JiraField:
    """Represents a Jira field with validation"""
    name: str
    field_id: str
    required: bool = False
    allowed_values: Optional[List[str]] = None
    
    def validate(self, value: any) -> bool:
        """Validate value against field constraints"""
        if self.required and value is None:
            raise ValueError(f"Field '{self.name}' is required")
        if self.allowed_values and value not in self.allowed_values:
            raise ValueError(
                f"Invalid value '{value}' for field '{self.name}'. "
                f"Allowed: {self.allowed_values}"
            )
        return True

class EnumCache:
    """Cache for Jira enumerations to reduce API calls"""
    
    def __init__(self, jira_client: JIRA):
        self.jira = jira_client
        self._cache: Dict[str, any] = {}
        self._field_cache: Dict[str, JiraField] = {}
    
    def get_issue_types(self, project_key: str) -> List[Dict]:
        """Get available issue types for project"""
        cache_key = f"issue_types_{project_key}"
        if cache_key not in self._cache:
            meta = self.jira.createmeta(projectKeys=project_key, expand='projects.issuetypes')
            self._cache[cache_key] = meta['projects'][0]['issuetypes'] if meta['projects'] else []
        return self._cache[cache_key]
    
    def get_statuses(self, project_key: str) -> List[Dict]:
        """Get available statuses"""
        cache_key = f"statuses_{project_key}"
        if cache_key not in self._cache:
            self._cache[cache_key] = self.jira.statuses()
        return self._cache[cache_key]
    
    def get_priorities(self) -> List[Dict]:
        """Get available priorities"""
        if 'priorities' not in self._cache:
            self._cache['priorities'] = self.jira.priorities()
        return self._cache['priorities']
    
    def get_field(self, field_name: str) -> Optional[JiraField]:
        """Get field metadata"""
        if field_name not in self._field_cache:
            fields = self.jira.fields()
            for f in fields:
                if f['name'] == field_name or f['id'] == field_name:
                    allowed = f.get('allowedValues', [])
                    self._field_cache[field_name] = JiraField(
                        name=f['name'],
                        field_id=f['id'],
                        allowed_values=[v.get('value', v.get('name')) for v in allowed] if allowed else None
                    )
                    break
        return self._field_cache.get(field_name)
    
    def clear_cache(self):
        """Clear all cached data"""
        self._cache.clear()
        self._field_cache.clear()

class JiraIssueManager:
    """Robust Jira issue manager with error handling and logging"""
    
    def __init__(self, server: str, username: str, password: str):
        self.logger = logging.getLogger("jira_client")
        self._connect(server, username, password)
        self.enum_cache = EnumCache(self.jira)
    
    @handle_jira_errors
    def _connect(self, server: str, username: str, password: str):
        """Establish connection with validation"""
        self.logger.info(f"Connecting to Jira server: {server}")
        self.jira = JIRA(
            server=server,
            basic_auth=(username, password),
            timeout=(30, 90)  # (connect, read) timeout in seconds
        )
        # Verify connection
        myself = self.jira.myself()
        self.logger.info(f"Connected as: {myself['displayName']}")
    
    @handle_jira_errors
    def update_issue(
        self,
        issue_key: str,
        summary: Optional[str] = None,
        description: Optional[str] = None,
        priority: Optional[Priority] = None,
        assignee: Optional[str] = None,
        custom_fields: Optional[Dict[str, any]] = None
    ) -> Dict:
        """
        Update a Jira issue with comprehensive error handling
        
        Args:
            issue_key: The issue identifier (e.g., 'PROJ-123')
            summary: New summary text
            description: New description
            priority: Priority enum value
            assignee: Username or email of assignee
            custom_fields: Dictionary of custom field updates
        
        Returns:
            Update result metadata
        """
        self.logger.info(f"Updating issue: {issue_key}")
        
        # Get current issue for validation
        try:
            issue = self.jira.issue(issue_key)
        except JIRAError:
            raise JiraNotFoundError(f"Issue {issue_key} not found")
        
        update_dict = {}
        
        # Validate and prepare updates
        if summary is not None:
            if not summary or len(summary) > 255:
                raise JiraValidationError("Summary must be 1-255 characters")
            update_dict['summary'] = summary
        
        if description is not None:
            update_dict['description'] = description
        
        if priority is not None:
            update_dict['priority'] = {'name': priority.name}
        
        if assignee is not None:
            update_dict['assignee'] = {'name': assignee}
        
        # Validate custom fields
        if custom_fields:
            for field_name, value in custom_fields.items():
                field_meta = self.enum_cache.get_field(field_name)
                if field_meta:
                    field_meta.validate(value)
                update_dict[field_meta.field_id if field_meta else field_name] = value
        
        # Perform update
        if update_dict:
            self.logger.debug(f"Update fields: {update_dict}")
            issue.update(fields=update_dict)
            self.logger.info(f"Successfully updated {issue_key}")
        
        return {
            'success': True,
            'issue_key': issue_key,
            'updated_fields': list(update_dict.keys()),
            'self': issue.self
        }
    
    @handle_jira_errors
    def transition_issue(
        self,
        issue_key: str,
        status: str,
        comment: Optional[str] = None,
        fields: Optional[Dict] = None
    ):
        """
        Transition issue to new status
        
        Args:
            issue_key: Issue identifier
            status: Target transition name
            comment: Optional transition comment
            fields: Optional fields to update during transition
        """
        self.logger.info(f"Transitioning {issue_key} to {status}")
        
        issue = self.jira.issue(issue_key)
        
        # Find transition ID
        transitions = self.jira.transitions(issue)
        transition_id = None
        for t in transitions:
            if t['name'].lower() == status.lower():
                transition_id = t['id']
                break
        
        if not transition_id:
            available = [t['name'] for t in transitions]
            raise JiraValidationError(
                f"Transition '{status}' not available. "
                f"Available: {available}"
            )
        
        # Execute transition
        kwargs = {'transition': transition_id}
        if comment:
            kwargs['comment'] = comment
        if fields:
            kwargs['fields'] = fields
        
        self.jira.transition_issue(issue, **kwargs)
        self.logger.info(f"Successfully transitioned {issue_key} to {status}")
    
    @handle_jira_errors
    def create_issue(
        self,
        project: str,
        summary: str,
        issue_type: IssueType,
        description: str = "",
        priority: Priority = Priority.MEDIUM,
        **kwargs
    ) -> Dict:
        """
        Create a new issue with full validation
        
        Args:
            project: Project key
            summary: Issue summary
            issue_type: Type enum
            description: Issue description
            priority: Priority enum
            **kwargs: Additional fields
        
        Returns:
            Created issue metadata
        """
        self.logger.info(f"Creating {issue_type.value} in {project}")
        
        # Validate project exists
        try:
            self.jira.project(project)
        except JIRAError:
            raise JiraNotFoundError(f"Project {project} not found")
        
        # Validate issue type is available
        available_types = self.enum_cache.get_issue_types(project)
        type_names = [t['name'].lower() for t in available_types]
        if issue_type.value.lower() not in type_names:
            raise JiraValidationError(
                f"Issue type '{issue_type.value}' not available. "
                f"Available: {[t['name'] for t in available_types]}"
            )
        
        issue_dict = {
            'project': {'key': project},
            'summary': summary,
            'description': description,
            'issuetype': {'name': issue_type.value},
            'priority': {'name': priority.name}
        }
        issue_dict.update(kwargs)
        
        new_issue = self.jira.create_issue(fields=issue_dict)
        self.logger.info(f"Created issue: {new_issue.key}")
        
        return {
            'key': new_issue.key,
            'id': new_issue.id,
            'self': new_issue.self
        }
