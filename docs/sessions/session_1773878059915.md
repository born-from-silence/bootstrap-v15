        current_dirs = directions.clone()

                current_origins, current_dirs, primary_density, primary_t

                hit_pos[hit_mask],
                hit_normal[hit_mask],

                    throughput.max(dim=-1, keepdim=True)[0]

        # Alpha compositing
        dists = t_values[..., 1:] - t_values[..., :-1]

            dim=-1
        )[:, :, :-1]
        
        weights = alpha * transmittance
        rendered = torch.sum(weights.unsqueeze(-1) * emitted, dim=-2)
        
        return rendered
